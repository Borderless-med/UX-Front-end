BACKEND (Supabase + Vercel Functions)
─────────────────────────────────────
□ Database Table: ai_scans
  CREATE TABLE ai_scans (
    id SERIAL PRIMARY KEY,
    scan_id VARCHAR(50) UNIQUE NOT NULL,
    user_session VARCHAR(255),
    status VARCHAR(20),  -- 'initiated', 'returned', 'booked'
    initiated_at TIMESTAMP DEFAULT NOW(),
    returned_at TIMESTAMP,
    scan_results JSONB,
    INDEX idx_scan_id (scan_id),
    INDEX idx_user_session (user_session)
  );

□ API Endpoint 1: POST /api/scans/initiate
  Input: { user_session: string }
  Output: { scan_id: "OL-2026-4521", redirect_url: "https://orachope.orallink.health?ref=OL-2026-4521" }
  Logic: Generate unique ID, store in DB, return redirect URL

□ API Endpoint 2: GET /api/scans/:scanId
  Input: scanId in URL
  Output: { valid: true/false, initiated_at: timestamp, status: string }
  Logic: Lookup in DB, return if exists

□ API Endpoint 3: POST /api/scans/:scanId/return
  Input: { scan_id: string, user_session: string }
  Output: { success: true, message: "Welcome back!" }
  Logic: Update status to 'returned', set returned_at timestamp
